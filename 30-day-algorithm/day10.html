        if (height == null || height.length <= 2) {
            return 0;
        }

        int left = 0;
        int right = height.length - 1;
        int leftMax = 0;

                }
                left++;
            } else {
                if (height[right] >= rightMax) {

            }
        }

        return water;
    }
