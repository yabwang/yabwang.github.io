import java.util.LinkedList;
import java.util.Queue;

class MyStack {
    private Queue<Integer> q1;
    private Queue<Integer> q2;

    public MyStack() {
        q1 = new LinkedList<>();
        q2 = new LinkedList<>();
    }

    // ä¿è¯æ–°å…ƒç´ åœ¨é˜Ÿå¤´ï¼ˆæ ˆé¡¶ï¼‰ï¼šæŠŠæ–°å…ƒç´ æ”¾å…¥ç©ºé˜Ÿåˆ—ï¼Œå†æŠŠå¦ä¸€é˜Ÿåˆ—å…¨éƒ¨ç§»è¿‡æ¥
    public void push(int x) {
        if (q1.isEmpty()) {
            q1.offer(x);
            while (!q2.isEmpty()) {
                q1.offer(q2.poll());
            }
        } else {
            q2.offer(x);
            while (!q1.isEmpty()) {
                q2.offer(q1.poll());
            }
        }
    }

    public int pop() {
        if (!q1.isEmpty()) {
            return q1.poll();
        }
        return q2.poll();
    }

    public int top() {
        if (!q1.isEmpty()) {
            return q1.peek();
        }
        return q2.peek();
    }

    public boolean empty() {
        return q1.isEmpty() && q2.isEmpty();
    }
}
